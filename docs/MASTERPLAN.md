# 22. ANHANG: DATENBANK-SCHEMA

## 22.1 Users

```sql
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email VARCHAR(255) UNIQUE NOT NULL,
  username VARCHAR(50) UNIQUE NOT NULL,
  tier VARCHAR(20) DEFAULT 'free', -- free, pro, pro_plus
  tier_expires_at TIMESTAMP,
  email_verified BOOLEAN DEFAULT FALSE,
  avatar_url VARCHAR(500),
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
```

## 22.2 Profiles

```sql
CREATE TABLE profiles (
  id UUID PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
  display_name VARCHAR(100),
  bio TEXT,
  discord VARCHAR(100),
  telegram VARCHAR(100),
  twitter VARCHAR(100),
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
```

## 22.3 Signals

```sql
CREATE TABLE signals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  ticker VARCHAR(20) NOT NULL,
  type VARCHAR(10) NOT NULL, -- BUY, SELL
  entry_price DECIMAL(20, 10) NOT NULL,
  stop_loss DECIMAL(20, 10) NOT NULL,
  tp1 DECIMAL(20, 10),
  tp2 DECIMAL(20, 10),
  tp3 DECIMAL(20, 10),
  tp4 DECIMAL(20, 10),
  tp5 DECIMAL(20, 10),
  confidence INTEGER, -- 0-100
  status VARCHAR(20) DEFAULT 'active', -- active, closed, cancelled
  result VARCHAR(20), -- win, loss, breakeven
  profit_percent DECIMAL(10, 2),
  claude_analysis TEXT,
  gemini_analysis TEXT,
  gpt_analysis TEXT,
  created_at TIMESTAMP DEFAULT NOW(),
  closed_at TIMESTAMP
);
```

## 22.4 Signal Trails

```sql
CREATE TABLE signal_trails (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  signal_id UUID REFERENCES signals(id) ON DELETE CASCADE,
  trail_number INTEGER NOT NULL,
  old_stop_loss DECIMAL(20, 10),
  new_stop_loss DECIMAL(20, 10),
  old_tps JSONB,
  new_tps JSONB,
  reason TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
```

## 22.5 Chat Messages

```sql
CREATE TABLE chat_messages (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  room_id VARCHAR(100) NOT NULL, -- 'global' or ticker like 'NACHO'
  user_id UUID REFERENCES users(id) ON DELETE SET NULL,
  username VARCHAR(50) NOT NULL,
  content TEXT NOT NULL,
  is_deleted BOOLEAN DEFAULT FALSE,
  deleted_by UUID REFERENCES users(id),
  created_at TIMESTAMP DEFAULT NOW()
);
```

## 22.6 Token News

```sql
CREATE TABLE token_news (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  ticker VARCHAR(20) NOT NULL,
  title VARCHAR(255) NOT NULL,
  content TEXT,
  source_url VARCHAR(500),
  source_type VARCHAR(20), -- website, discord, telegram, twitter
  is_verified BOOLEAN DEFAULT FALSE,
  verified_by UUID REFERENCES users(id),
  verified_at TIMESTAMP,
  crawled_at TIMESTAMP DEFAULT NOW()
);
```

## 22.7 API Keys

```sql
CREATE TABLE api_keys (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  key_hash VARCHAR(255) NOT NULL, -- Hashed key
  key_prefix VARCHAR(20) NOT NULL, -- knx_live_ or knx_test_
  name VARCHAR(100),
  tier VARCHAR(50) NOT NULL, -- kaspa, kaspa_krc20, signal, complete
  requests_today INTEGER DEFAULT 0,
  requests_month INTEGER DEFAULT 0,
  last_used_at TIMESTAMP,
  expires_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);
```

## 22.8 Sponsors

```sql
CREATE TABLE sponsors (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  logo_url VARCHAR(500),
  website_url VARCHAR(500),
  placement VARCHAR(50), -- homepage, sidebar, footer
  is_active BOOLEAN DEFAULT TRUE,
  display_order INTEGER DEFAULT 0,
  starts_at TIMESTAMP,
  ends_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);
```
