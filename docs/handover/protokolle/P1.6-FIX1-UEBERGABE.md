```tsx
interface FeatureBannerProps {
  // ...existing props
  compact?: boolean;
}

// Compact layout fuer Grid-Ansicht

  return (
    <Link className="...h-full">
      <div className="flex flex-col h-full">
        <div className="w-10 h-10 ...mb-3">{icon}</div>
        <h3 className="text-base ...">{title}</h3>
        <p className="text-xs ...line-clamp-2">{description}</p>
        <div className="...text-xs">{cta}</div>
      </div>

  );
}
```

```tsx
interface InfraItem {
  // ...existing fields
  logo?: string;
}

// Logo oder Buchstaben-Fallback

  <Image src={item.logo} ... />

    <span>{item.name.charAt(0)}</span>
  </div>
)}
```
